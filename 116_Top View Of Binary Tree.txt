#include <bits/stdc++.h> 
/************************************************************

    Following is the TreeNode class structure:

    template <typename T>
    class TreeNode {
       public:
        T val;
        TreeNode<T> *left;
        TreeNode<T> *right;

        TreeNode(T val) {
            this->val = val;
            left = NULL;
            right = NULL;
        }
    };

************************************************************/

vector<int> getTopView(TreeNode<int> *root) {
    if(!root)   return {};
    vector<int> v;
    map<int, int> mp;
    queue<pair<TreeNode<int>*, int> > q;
    q.push({root, 0});
        
    while(!q.empty())   {
        pair<TreeNode<int>*, int> p = q.front();
        q.pop();
        if(mp.find(p.second) == mp.end())
        mp[p.second] = p.first->val;
        if(p.first->left)   q.push({p.first->left, p.second-1});
        if(p.first->right)   q.push({p.first->right, p.second+1});
    }
        
    for(auto it : mp)  v.push_back(it.second);
    return v;
}
